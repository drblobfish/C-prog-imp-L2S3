\subsection*{a}

\begin{lstlisting}
int factorielle(int n){
    int sum = 1;
    for (int i = n;i>1;i--){
        sum *= i;
    }
    return sum;
}
\end{lstlisting}

\subsection*{b}

\begin{lstlisting}

int factorielle(int n){
    if (n == 1){
        return n;
    }
    return n * factorielle(n-1);
}
\end{lstlisting}
